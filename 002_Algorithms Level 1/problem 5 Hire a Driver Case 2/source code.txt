 static void Main(string[] args)
        {
            System.Console.WriteLine(isHired(15, false, true) ? "Hired" : "Rejected");
        }
        
static bool isHired(int age, bool hasCertificate, bool hasRecommendation)
        {
            return (age > 21 && hasCertificate) || hasRecommendation ? true : false;
        }
