static double penny(Currency currency)
        {
            return currency.Penny +
             currency.Nickel * 5 +
             currency.Dime * 10 +
             currency.Quarter * 25 +
             currency.Dollar * 100;
        }

class Currency
        {
            public int Penny { get; set; }
            public int Nickel { get; set; }
            public int Dime { get; set; }
            public int Quarter { get; set; }
            public int Dollar { get; set; }

        }
