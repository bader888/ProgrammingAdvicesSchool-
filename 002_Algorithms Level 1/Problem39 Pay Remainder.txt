static void Main(string[] args)
        {
            Console.Write("Enter Total Bill: ");
            double totalBill = double.Parse(Console.ReadLine());

            Console.Write("Enter Cash Paid: ");
            double cashPaid = double.Parse(Console.ReadLine());

            Console.WriteLine(reminderBack(totalBill, cashPaid));
        }

static double reminderBack(double totalBill, double cashPaid)
        {
            return cashPaid - totalBill;
        }
