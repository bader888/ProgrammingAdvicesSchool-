static void Main(string[] args)
        {
            Console.Write("Enter Bill Vaue: ");
            double billValue = double.Parse(Console.ReadLine());

            Console.WriteLine(totalBill(billValue));
        }

static double totalBill(double billValue)
        {
            double totalBill = billValue * 1.1;
            totalBill = totalBill * 1.16;
            return totalBill;
        }
